>> n1 = -1:5; f1 = exp(n1).*n1.^2;		%1st case
>> n2 = 5:10; f2 = n2*0;			%2nd case
>> n3 = 10:15; f3 = [-3 -3 -3 -3 -3 -3];	%3rd case
>> n = [n1 n2 n3];				%n array
>> f = [f1 f2 f3];				%f array
>> stem(f,n)

>> n3 = 10:15; f3 = [-3000 -3000 -3000 -3000 -3000 -3000];
